import { Component } from '@angular/core';

@Component({
  selector: 'houses',
  template: `
  <h1>Houses</h1>
  <ul>
    <li *ngFor="let name of houses">
      <house [name]="name"></house>
    </li>
  </ul>
  `,
})
export class HousesComponent {
  private houses = ['Stark', 'Greyjoy', 'Bolton'];
}
