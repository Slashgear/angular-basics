### Exercise 2

Add input for changing project Name :

Tips -> Banana Box [()]
[property] (event)


```{TypeScript}
@Component({
    selector: 'my-app',
    template: `
      <h1>My First {{projectName}}</h1>
      <input type="text" [(ngModel)]="projectName"/>
    `
})
export class AppComponent {
  private projectName: string = "Hello world";
}
```
